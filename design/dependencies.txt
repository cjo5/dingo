type Decl interface {
    Symbol() *Symbol
    Context() *FileContext // Nil if not top level

    addDependency(Decl)
    dependencies() []Decl
    setColor(GraphColor)
    color() GraphColor
}

type baseDecl struct {
    Sym *Symbol
    Ctx *FileContext

    deps []Decl
    color graphColor
}

type Module struct {
    Decls []Decl
    Imports []Import
}

type Import struct {
    Ctx *FileContext
}
